#   py -3.11 -m venv C:\ClaudeCore\audio-venv
#   C:\ClaudeCore\audio-venv\Scripts\python -m pip install --upgrade pip
#   C:\ClaudeCore\audio-venv\Scripts\python -m pip install -r tools\audio-requirements.txt
#
# Then accept the gated license at https://huggingface.co/stabilityai/stable-audio-open-1.0,
# `hf auth login` with a read token, and start the server with tools\run-audio-server.ps1.
